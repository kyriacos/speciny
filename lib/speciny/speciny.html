Since Ruby 1.9 however a singleton_class was introduced which provides acess to the singleton of the object.

module Speciny
  class MatcherObject
    attr_reader :actual, :expected, :result
    def initialize(actual, expected=nil, &block)
      @actual = actual
      @expected = expected
      @result = self.matches? if !expected.nil?
    end

    def ==(other)
      @result = @actual == other
    end

    def matches?
      @expected.matches?(@actual)
    end
  end
end
